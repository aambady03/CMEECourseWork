# count the number of lines in FASTA file
wc -l ../data/fasta/*.fasta 

# print everything from line 2 for E.coli genome
tail -n +2 ../data/fasta/E.coli.fasta 

# count section of sequence
tail -n +2 ../data/fasta/E.coli.fasta  | tr -d '\n' | wc -c


# find matches to ATGC in e-coli
tail -n +2 ../data/fasta/E.coli.fasta  | tr -d '\n' | grep -oi ATCG | wc -l

# Extract sequence without header
SEQUENCE=$(tail -n +2 ../data/fasta/E.coli.fasta | tr -d '\n')

#find how many A & T bases. according to the Chargaff’s rule
#find how many C & G bases. according to the Chargaff’s rule  
#compute AT/GC ratio 
tail -n +2 ../data/fasta/E.coli.fasta | tr -d '\n' | grep -o '[A]' | wc -l 
tail -n +2 ../data/fasta/E.coli.fasta | tr -d '\n' | grep -o '[C]' | wc -l
